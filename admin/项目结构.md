### 后台目录结构

---



```
├── README.md			// 项目简介
├── build				// webpack配置文件
│   ├── build.js
│   ├── check-versions.js
│   ├── dev-client.js
│   ├── dev-server.js
│   ├── utils.js
│   ├── vue-loader.conf.js
│   ├── webpack.base.conf.js
│   ├── webpack.dev.conf.js
│   └── webpack.prod.conf.js
├── config				// 项目路径相关设置
│   ├── dev.env.js
│   ├── index.js
│   └── prod.env.js
├── index.html			// 入口html文件
├── package.json		// 项目依赖模块文件
├── src					// 源代码
│   ├── App.vue			//页面入口Vue文件
│   ├── assets			// 需要通过webpack打包的资源
│   │   ├── css
│   │   │   └── selectize.default.css
│   │   ├── images
│   │   │   ├── kangna.jpg
│   │   │   ├── kangna2.jpeg
│   │   │   └── threeline.png
│   │   └── js
│   │       ├── LineChart.js
│   │       ├── selectize.min.js
│   │       └── translateRoute.js
│   ├── components			// Vue组件
│   │   ├── Info.vue
│   │   └── common
│   ├── main.js				// 入口js文件
│   ├── plugins				// 插件
│   │   └── selectize.js
│   ├── router				// 路由
│   │   └── index.js
│   ├── store			// vuex,状态管理中心
│   │   ├── actions.js
│   │   ├── getters.js
│   │   ├── index.js
│   │   ├── mutations.js
│   │   └── state.js
│   └── views				// 页面视图文件
│       ├── Index.vue		// 首页
│       ├── animal			// 双层目录示例
│       │   ├── Index.vue
│       │   ├── Name.vue
│       │   ├── Size.vue
│       │   └── people
│       │       ├── Age.vue
│       │       ├── Index.vue
│       │       └── Name.vue
│       ├── layout  		// 总布局
│       │   ├── Layout.vue
│       │   ├── Main.vue
│       │   ├── NavBar.vue
│       │   └── SideBar.vue
│       └── store			// 三层目录示例
│           ├── Index.vue
│           ├── storeInfoMaintain
│           │   ├── ConnectPlatform.vue
│           │   ├── Index.vue
│           │   └── MallAnnouncement.vue
│           └── storeManage
│               ├── Index.vue
│               ├── ReviewStore.vue
│               └── StoreList.vue
├── static			// 不需要经过webpack打包的静态资源
```
